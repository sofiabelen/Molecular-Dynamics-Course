\section{Время динамической памяти \(t_m \approx 3\)}

\begin{equation}
    <\Delta r^2(t)> = \frac{1}{N} \sum_{i = 1}^N (\mathbf{r}_i(t) - \mathbf{r}_i'(t))^2
\end{equation}

\begin{equation}
    <\Delta v^2(t)> = \frac{1}{N} \sum_{i = 1}^N (\mathbf{v}_i(t) - \mathbf{v}_i'(t))^2
\end{equation}
